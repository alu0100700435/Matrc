def initialize(num, den)                
  com = gcd(num, den)
  if (den < 0)
    @num, @den = ((-1*num)/com), ((-1*den)/com)
  else
    @num, @den = num/com, den/com
  end
end

def gcd(u, v)
        u, v = u.abs, v.abs
        while v != 0
                u, v = v, u % v
        end
        u
end
